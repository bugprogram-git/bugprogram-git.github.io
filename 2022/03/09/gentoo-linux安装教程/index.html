#设置主机名
nano /etc/hostname

#设置root用户密码
passwd root

#安装一些基本的网络管理软件包
emerge --ask net-wireless/iw net-wireless/wpa_supplicant net-misc/networkmanager net-misc/dhcp net-misc/dnsmasq

#如果是当服务器可以设置静态ip地址
nano /etc/systemd/network/10-wired-dhcp.network(数字代表优先级(越小越优先)

#静态网络配置
[Match]
Name=eth0

[Network]
Address=192.168.50.100/24
DNS=192.168.50.1
DNS=114.114.114.114

[Route]
Gateway=192.168.50.1
Metric=100
#设置网络服务开机自启动
ln -snf /run/systemd/resolve/resolv.conf /etc/resolv.conf
systemctl enable systemd-resolved.service
systemctl enable systemd-networkd.service
